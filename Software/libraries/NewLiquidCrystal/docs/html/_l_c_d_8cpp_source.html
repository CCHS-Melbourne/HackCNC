00072 void LCD::setCursor(uint8_t col, uint8_t row)
00073 {
00074    int row_offsetsDef[]   = { 0x00, 0x40, 0x14, 0x54 }; // For regular LCDs
00075    int row_offsetsLarge[] = { 0x00, 0x40, 0x10, 0x50 }; // For 16x4 LCDs
00076    
00077    if ( row >= _numlines ) 
00078    {
00079       row = _numlines-1;    // rows start at 0
00080    }
00081    
00082    // 16x4 LCDs have special memory map layout
00083    // ----------------------------------------
00084    if ( _cols == 16 && _numlines == 4 )
00085    {
00086       command(LCD_SETDDRAMADDR | (col + row_offsetsLarge[row]));
00087    }
00088    else 
00089    {
00090       command(LCD_SETDDRAMADDR | (col + row_offsetsDef[row]));
00091    }
00092 
00093 }
